1. Get the application URL by running these commands:
{{- if .Values.es.ingress.enabled }}
{{- range .Values.es.ingress.hosts }}
  http{{ if $.Values.es.ingress.tls }}s{{ end }}://{{ . }}{{ $.Values.es.ingress.path }}
{{- end }}
{{- else if contains "NodePort" .Values.es.service.type }}
  export NODE_PORT=$(kubectl get -o jsonpath="{.spec.ports[0].nodePort}" services {{ template "log-factory-es.fullname" . }})
  export NODE_IP=$(kubectl get nodes -o jsonpath="{.items[0].status.addresses[0].address}")
  echo http://$NODE_IP:$NODE_PORT
{{- else if contains "LoadBalancer" .Values.es.service.type }}
     NOTE: It may take a few minutes for the LoadBalancer IP to be available.
           You can watch the status of by running 'kubectl get svc -w {{ template "log-factory-es.fullname" . }}'
  export SERVICE_IP=$(kubectl get svc {{ template "log-factory-es.fullname" . }} -o jsonpath='{.status.loadBalancer.ingress[0].ip}')
  echo http://$SERVICE_IP:{{ .Values.es.service.ports.db.port }}
{{- else if contains "ClusterIP" .Values.es.service.type }}
  export POD_NAME=$(kubectl get pods -l "app={{ template "log-factory-es.name" . }},release={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
  echo "Visit http://127.0.0.1:8080 to use your application"
  kubectl port-forward $POD_NAME 8080:80
{{- end }}
